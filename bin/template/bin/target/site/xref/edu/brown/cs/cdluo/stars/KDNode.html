1   package edu.brown.cs.cdluo.stars;
2   
3   import java.util.Objects;
4   
5   /**
6    * Represents a node in the KDTree by keeping track of a star along with a left
7    * and right child.
8    * @author cdluo
9    *
10   */
11  public class KDNode {
12    private Star data;
13    private KDNode left;
14    private KDNode right;
15  
16    /**
17     * Constructor takes the star to use as the data in this node.
18     * @param star
19     *          the star to be used as data.
20     */
21    public KDNode(Star star) {
22      data = star;
23      left = null;
24      right = null;
25    }
26  
27    /**
28     * Sets the left child of this node.
29     * @param l
30     *          the node to be set
31     */
32    public void setLeft(KDNode l) {
33      left = l;
34    }
35  
36    /**
37     * Sets the right child of this node.
38     * @param r
39     *          the node to be set
40     */
41    public void setRight(KDNode r) {
42      right = r;
43    }
44  
45    /**
46     * Gets the left child of this node.
47     * @return the left child
48     */
49    public KDNode getLeft() {
50      return left;
51    }
52  
53    /**
54     * Gets the right child of this node.
55     * @return the left child
56     */
57    public KDNode getRight() {
58      return right;
59    }
60  
61    /**
62     * Gets the Star of this node.
63     * @return the Star
64     */
65    public Star getData() {
66      return data;
67    }
68  
69    @Override
70    public String toString() {
71      return data.toString() + "|Left: " + left.toString() + "|Right: "
72          + right.toString();
73    }
74    /**
75     * Checks for equality.
76     * @param kd2 the KDNode to compare to.
77     * @return boolean
78     */
79    public boolean equals(KDNode kd2) {
80      if (data.equals(kd2.getData())) {
81        if (right.equals(kd2.getRight())) {
82          if (left.equals(kd2.getLeft())) {
83            return true;
84          }
85        }
86      }
87      return false;
88    }
89  
90    /**
91     * Returns a hashcode of this class.
92     * @return a hashcode of this class.
93     */
94    public int hashCode() {
95      return Objects.hash(data, left, right);
96    }
97  }
