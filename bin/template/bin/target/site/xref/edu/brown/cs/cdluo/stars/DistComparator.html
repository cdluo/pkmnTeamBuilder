3   import java.util.Comparator;
4   
5   /**
6    * Comparator for distances in DQEntry.
7    * @author cdluo
8    *
9    */
10  public class DistComparator implements Comparator<DQEntry> {
11  
12    /**
13     * Compares two DQEntries by their distances.
14     * @param e1
15     *          1st Entry to be compared
16     * @param e2
17     *          2nd Entry to be compared
18     * @return an int for the compare method.
19     */
20    @Override
21    public int compare(DQEntry e1, DQEntry e2) {
22      // First check for null values (happens if DistQueue was just instantiated.)
23      if (e1 == null && e2 == null) {
24        return 0;
25      } else if (e1 == null) {
26        return 1;
27      } else if (e2 == null) {
28        return -1;
29      }
30      // Now compare actual values
31      if (e1.getDist() > e2.getDist()) {
32        return 1;
33      } else if (e1.getDist() == e2.getDist()) {
34        return 0;
35      } else {
36        return -1;
37      }
38    }
39  }
