3   import java.util.Objects;
4   
5   /**
6    * This class represents a star by holding all the information from the csv
7    * file.
8    * @author cdluo
9    */
10  public class Star {
11    private String starID;
12    private String properName;
13    private Dimen dimen;
14  
15    /**
16     * Constructs the star with 5 parameters from the csv file.
17     * @param id
18     *          the id of the star
19     * @param name
20     *          its name
21     * @param x
22     *          and the 3 coordinates
23     * @param y
24     *          ^
25     * @param z
26     *          ^
27     */
28    public Star(String id, String name, double x, double y, double z) {
29      starID = id;
30      properName = name;
31      double[] input = {x, y, z};
32      dimen = new Dimen(input);
33    }
34  
35    /**
36     * Prints all 5 global variables stored in the star.
37     * @return String
38     */
39    @Override
40    public String toString() {
41      return String.format("%s|%s|%.2f|%.2f|%.2f", starID, properName,
42          dimen.getDimenX(0), dimen.getDimenX(1), dimen.getDimenX(2));
43    }
44  
45    /**.
46     * Accessor for the dimen
47     * @return dimen
48     */
49    public Dimen getDimen() {
50      return dimen;
51    }
52  
53    /**.
54     * Accessor for the ID
55     * @return String
56     */
57    public String getID() {
58      return starID;
59    }
60  
61    /**.
62     * Accessor for the name
63     * @return name
64     */
65    public String getName() {
66      return properName;
67    }
68  
69    /**.
70     * Checks for equality
71     * @param s2 the star to compare to.
72     * @return boolean
73     */
74    public boolean equals(Star s2) {
75      if (s2.dimen.equals(dimen)) {
76        if (s2.properName.equals(properName)) {
77          if (s2.starID.equals(starID)) {
78            return true;
79          }
80        }
81      }
82      return false;
83    }
84  
85    /**
86     * Returns a hashcode of this class.
87     * @return a hashcode of this class.
88     */
89    public int hashCode() {
90      return Objects.hash();
91    }
92  }
