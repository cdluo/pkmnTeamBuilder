30    /**
31     * The constructor which builds the tree.
32     * @param stars
33     *          an array of stars to build the tree with.
34     */
35    public KDTree(Star[] stars) {
36      // Can take in anything with a Dimen object
37      buildTree(stars, 0);
38    }

40    /**
41     * Recursively builds the KDTree using the stars array provided.
42     * @param stars
43     *          the provided stars
44     * @param curDimen
45     *          the current dimen we are building on/depth of the tree.
46     * @return the root of the tree (at the top call).
47     */
48    public KDNode buildTree(Star[] stars, int curDimen) {
49      sortByDimen(stars, curDimen);
50      int mid = stars.length / 2;
51  
52      if (empty) {
53        addRoot(stars[mid]);
54        empty = false;
55        Star[] firstHalf = Arrays.copyOfRange(stars, 0, mid);
56        Star[] secondHalf = Arrays.copyOfRange(stars, mid + 1, stars.length);
57        root.setLeft(buildTree(firstHalf, incDimen(curDimen)));
58        root.setRight(buildTree(secondHalf, incDimen(curDimen)));
59        return root;
60      } else if (stars.length == 0 || stars.length == 1) { // Base case
61        if (stars.length == 0) {
62          return null;
63        } else {
64          KDNode node = new KDNode(stars[0]);
65          return node;
66        }
67      } else { // General Case
68        KDNode n = new KDNode(stars[mid]);
69        Star[] fHalf = Arrays.copyOfRange(stars, 0, mid);
70        Star[] sHalf = Arrays.copyOfRange(stars, mid + 1, stars.length);
71        n.setLeft(buildTree(fHalf, incDimen(curDimen)));
72        n.setRight(buildTree(sHalf, incDimen(curDimen)));
73        return n;
74      }
75    }

78     * Sorts a stars array by a dimension.
79     * @param stars
80     *          the array to be sorted.
81     * @param cDimen
82     *          the dimen we're sorting on.

84    public void sortByDimen(Star[] stars, int cDimen) {
85      StarComparator sCompare = new StarComparator(cDimen);
86      Arrays.sort(stars, sCompare);

95      KDNode r = new KDNode(s);
96      root = r;

100   /**
101    * Returns the current dimension +1, unless it would be the max dimension, in
102    * which case it returns 0 to revert back to checking the first index of a
103    * Dimen object.
104    * @param curDimen
105    *          an int representing a dimension that a method is checking on
106    * @return an int representing the next dimension.
107    */
108   public int incDimen(int curDimen) {
109     if (curDimen + 1 == DIMEN) {
110       return 0;
111     } else {
112       return curDimen + 1;
113     }
114   }

116   // knn//
117   /**
118    * This prep method instantiates the resources needed for knn. By making these
119    * resources global, I don't have to pass them all as arguments with each
120    * recursive call of knn.
121    * @param k
122    *          the number of neighbors to find
123    * @param point
124    *          the point we're looking for neighbors around
125    * @param nameArg
126    *          whether the user input a name or not instead of a coordinate.
127    */
128   public void prepknn(int k, Dimen point, boolean nameArg) {
129     if (nameArg) { // If there is a name arg, I won't print the first element,
130       k++; // so I need to search an extra neighbor.
131     }
132     dq = new DistQueue(k, point);
133     knnPoint = point;
134   }
135 
136   /**
137    * A recursive knn search, which stores the nearest neighbors in the global
138    * distqueue (dq).
139    * @param curr
140    *          the current Node we're on/ the next one to search in calls.
141    * @param curDimen
142    *          the dimension we're sorting/searching on.
143    * @return A distqueue holding the nearest neighbors.
144    */
145   public DistQueue knn(KDNode curr, int curDimen) {
146     if (curr == null) {
147       return dq;
148     }
149 
150     dq.insert(curr.getData());
151 
152     double pointCurDimen = knnPoint.getDimenX(curDimen); // curDimen value of
153                                                          // the point we're
154                                                          // searching for.
155     double curCurDimen = curr.getData().getDimen().getDimenX(curDimen);
156     if (pointCurDimen < curCurDimen) {
157       knn(curr.getLeft(), incDimen(curDimen));
158     } else {
159       knn(curr.getRight(), incDimen(curDimen));
160     }
161     // Walking back up the tree to check if we should examine the other subtree.
162     if (!dq.isFull() || Math.abs(curCurDimen - pointCurDimen)
163         < dq.getMaxDist()) {
164       // Reversed searching left or right to cover the other subtree.
165       if (pointCurDimen < curCurDimen) {
166         knn(curr.getRight(), incDimen(curDimen));
167       } else {
168         knn(curr.getLeft(), incDimen(curDimen));
169       }
170     }
171     return dq;
172   }

175   /**
176    * Prepares the radius search similarly to prepknn.
177    * @param r
178    *          the radius to search in.
179    * @param point2
180    *          the point to search around.
181    */
182   public void prepRadius(double r, Dimen point2) {
183     radPoint = point2;
184     radius = r;
185     radList = new ArrayList<DQEntry>();
186   }

188   /**
189    * Recursive radius search. It's almost the same as knn, but with some minor
190    * changes noted in the comments.
191    * @param curr
192    *          As with knn the current/next node.
193    * @param curDimen
194    *          ^
195    * @return An ArrayList holding the Entries which hold the stars in the
196    *         radius.
197    */
198   public ArrayList<DQEntry> radius(KDNode curr, int curDimen) {
199     // I decided to make an arraylist of DQEntries since I already had a sorting
200     // comparator
201     // for them (see next method.)
202     if (curr == null) {
203       return radList;
204     }
205 
206     // This time, only insert if the distance is within the radius.
207     double currToPoint = radPoint.calcDist(curr.getData().getDimen());
208     if (currToPoint <= radius) {
209       DQEntry dqe = new DQEntry(curr.getData(), currToPoint);
210       radList.add(dqe);
211     }
212 
213     double pointCurDimen = radPoint.getDimenX(curDimen);
214     double curCurDimen = curr.getData().getDimen().getDimenX(curDimen);
215 
216     if (pointCurDimen < curCurDimen) {
217       radius(curr.getLeft(), incDimen(curDimen));
218     } else {
219       radius(curr.getRight(), incDimen(curDimen));
220     }
221 
222     // Walking back up the tree
223     // For radius, we don't need to worry about the queue not being full. Just
224     // check
225     // if the other subtree is viable for searching.
226     if (Math.abs(curCurDimen - pointCurDimen) < radius) {
227       if (pointCurDimen < curCurDimen) {
228         radius(curr.getRight(), incDimen(curDimen));
229       } else {
230         radius(curr.getLeft(), incDimen(curDimen));
231       }
232     }
233     return radList;
234   }
