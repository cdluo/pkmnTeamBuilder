74    /**
75     * The main execution method. Reads command line args to see if a gui should
76     * be launched. Also contains a REPL for when --gui is not specified.
77     * @throws IOException
78     */
79    private void run() throws IOException {
80      OptionParser parser = new OptionParser();
81  
82      parser.accepts("gui");
83      OptionSpec<File> fileSpec = parser.nonOptions().ofType(File.class);
84      OptionSet options = parser.parse(args);
85  
86      db = options.valueOf(fileSpec);
87      if (db == null) {
88        System.out.println("ERROR: Please specify a star file");
89        System.exit(1);
90      }
91  
92      stars = new ArrayList<Star>();
93      Scanner scanner = new Scanner(db);
94      scanner.useDelimiter(",|\\n");
95  
96      for (int x = 0; x < HEADER_SIZE; x++) { // Skip the header
97        String header = scanner.next();
98        if (x == ID_CORD && !header.equals("StarID")) {
99          System.out.println("ERROR: CSV Header has " + '"' + header + '"'
100             + " instead of: \"StarID\"");
101         System.exit(1);
102       }
103       if (x == NAME_CORD && !header.equals("ProperName")) {
104         System.out.println("ERROR: CSV Header has " + '"' + header + '"'
105             + " instead of: \"ProperName\"");
106         System.exit(1);
107       }
108       if (x == X_CORD && !header.equals("X")) {
109         System.out.println("ERROR: CSV Header has " + '"' + header + '"'
110             + " instead of: \"X\"");
111         System.exit(1);
112       }
113       if (x == Y_CORD && !header.equals("Y")) {
114         System.out.println("ERROR: CSV Header has " + '"' + header + '"'
115             + " instead of: \"Y\"");
116         System.exit(1);
117       }
118       if (x == Z_CORD && !header.equals("Z")) {
119         System.out.println("ERROR: CSV Header has " + '"' + header + '"'
120             + " instead of: \"Z\"");
121         System.exit(1);
122       }
123     }
124 
125     while (scanner.hasNext()) {
126       String[] starArgs = new String[HEADER_SIZE];
127       for (int i = 0; i < HEADER_SIZE; i++) {
128         starArgs[i] = scanner.next();
129       }
130       if (!starArgsCheck(starArgs)) {
131         System.out.println("There are formatting errors. "
132             + "Please fix them and resubmit.");
133         System.exit(1);
134       }
135       double x = Double.parseDouble(starArgs[X_CORD]);
136       double y = Double.parseDouble(starArgs[Y_CORD]);
137       double z = Double.parseDouble(starArgs[Z_CORD]);
138       Star star = new Star(starArgs[0], starArgs[1], x, y, z);
139       stars.add(star);
140     }
141 
142     scanner.close();
143     // Star data has been stored in Arraylist stars.
144 
145     // No more adding to stars, so convert to array.
146     tree = new KDTree(stars.toArray(new Star[stars.size()]));
147 
148     if (options.has("gui")) {
149       runSparkServer();
150     } else {
151       // ///The REPL//////
152       while (true) {
153         String args2 = null;
154         BufferedReader buff = new BufferedReader(new InputStreamReader(
155             System.in));
156         args2 = buff.readLine();
157         if (args2 == null || args2.length() == 0) {
158           break; // ctrl-d or empty line found
159         }
160         String args2NameChecked = prepInput(args2); // Checking for " " in
161                                                     // names.
162 
163         StringTokenizer st = new StringTokenizer(args2NameChecked, " ");
164         int inputSize = st.countTokens();
165         if (inputSize != ARG_CORD && inputSize != ARG_NAME) {
166           System.out.println("ERROR: Incorrect # of args.");
167           continue;
168         }
169 
170         String command = st.nextToken();
171 
172         if (command.equals("neighbors")) {
173           int k = Integer.parseInt(st.nextToken());
174           if (k == 0) { // Not looking for neighbors.
175             System.out.println(args2);
176             continue;
177           }
178 
179           if (inputSize == ARG_CORD) {
180             double x = Double.parseDouble(st.nextToken());
181             double y = Double.parseDouble(st.nextToken());
182             double z = Double.parseDouble(st.nextToken());
183             System.out.println(args2);
184             if (!neighbors(k, x, y, z).equals("")) {
185               System.out.println(neighbors(k, x, y, z));
186             }
187           } else if (inputSize == ARG_NAME) {
188             String name = st.nextToken();
189             String revertName = name.replaceAll("/////", " ");
190             System.out.println(args2); // Print original args2
191             if (!neighborsName(k, revertName).equals("")) {
192               System.out.println(neighborsName(k, revertName));
193             }
194           } else {
195             System.out.println("Incorrect # of args for neighbors.");
196             continue;
197           }
198         } else if (command.equals("radius")) {
199           double r = Double.parseDouble(st.nextToken());
200 
201           if (inputSize == ARG_CORD) {
202             double x = Double.parseDouble(st.nextToken());
203             double y = Double.parseDouble(st.nextToken());
204             double z = Double.parseDouble(st.nextToken());
205             System.out.println(args2);
206             if (!radius(r, x, y, z).equals("")) {
207               System.out.println(radius(r, x, y, z));
208             }
209           } else if (inputSize == ARG_NAME) {
210             String name = st.nextToken();
211             String revertName = name.replaceAll("/////", " ");
212             System.out.println(args2);
213             if (!radiusName(r, revertName).equals("")) {
214               System.out.println(radiusName(r, revertName));
215             }
216           } else {
217             System.out.println("Incorrect # of args for radius.");
218             continue;
219           }
220         } else {
221           System.out.println(args2);
222           System.out
223               .println("ERROR: Please use command 'neighbors' or 'radius'");
224         }
225       }
226     }
227   }

229   /**
230    * Checks if the CSV is giving proper star information.
231    * @param starArgs
232    *          the args to check (should be 5 at a time)
233    * @return boolean indicating if we can proceed (input is good.)
234    */
235   public boolean starArgsCheck(String[] starArgs) {
236     boolean goodToGo = true;
237     try {
238       Double.parseDouble(starArgs[ID_CORD]);
239     } catch (Exception e) {
240       System.out.println("ERROR: " + starArgs[ID_CORD]
241           + " is not a numerical ID.");
242       goodToGo = false;
243     }
244 
245     // No need to check starArgs[1], that's the name which can be anything.
246 
247     try {
248       Double.parseDouble(starArgs[X_CORD]);
249     } catch (Exception e) {
250       System.out.println("ERROR: " + starArgs[X_CORD] + " is not a double.");
251       goodToGo = false;
252     }
253 
254     try {
255       Double.parseDouble(starArgs[Y_CORD]);
256     } catch (Exception e) {
257       System.out.println("ERROR: " + starArgs[Y_CORD] + " is not a double.");
258       goodToGo = false;
259     }
260 
261     try {
262       Double.parseDouble(starArgs[Z_CORD]);
263     } catch (Exception e) {
264       System.out.println("ERROR: " + starArgs[Z_CORD] + " is not a double.");
265       goodToGo = false;
266     }
267 
268     return goodToGo;
269   }

304   public String neighborsName(int k, String name) {
305     String printedResult = "";
306     DistQueue result;
307     Dimen point = null;
308     for (Star s : stars) {
309       String formattedName = '"' + s.getName() + '"';
310       if (formattedName.equals(name)) {
311         point = s.getDimen();
312       }
313     }
314     tree.prepknn(k, point, true);
315     result = tree.knn(tree.getRoot(), 0);
316     printedResult = result.toString2();
317     return printedResult;
318   }

321    * Returns a string version of the radius search result.
322    * @param r
323    *          the radius to search in
324    * @param x
325    *          coordinate
326    * @param y
327    *          coordinate
328    * @param z
329    *          coordinate
330    * @return A string
331    */
332   public String radius(double r, double x, double y, double z) {
333     ArrayList<DQEntry> result;
334     String printedResult = "";
335     double[] input = {x, y, z};
336     Dimen point = new Dimen(input);
337     tree.prepRadius(r, point);
338     result = tree.sortRadius(tree.radius(tree.getRoot(), 0));
339     for (int i = 0; i < result.size(); i++) {
340       printedResult += result.get(i).toString();
341       if (i + 1 != result.size()) {
342         printedResult += "\n";
343       }
344     }
345     return printedResult;
346   }

378   /**
379    * If a space is found between quotes, replaces that space with "/////" to
380    * allow name reading for searches. (///// will be converted back)
381    * @param arg
382    *          the argument string to check
383    * @return the string according to specifications listed above.
384    */
385   public String prepInput(String arg) {
386     int firstQuote = arg.indexOf('"');
387     if (firstQuote != -1) {
388       int secondQuote = arg.indexOf('"', firstQuote + 1);
389       if (secondQuote != -1) {
390         String left = arg.substring(0, firstQuote);
391         String offender = arg.substring(firstQuote, secondQuote + 1);
392         String right = arg.substring(secondQuote + 1, arg.length());
393         String formattedOffender = offender.replaceAll(" ", "/////");
394         return left + formattedOffender + right;
395       }
396     }
397     return arg;
398   }

448   /**
449    * Handles the knn results page.
450    * @return a modelandview object representing the results page.
451    * @author cdluo
452    */
453   private class NNHandler implements TemplateViewRoute {
454     public ModelAndView handle(Request req, Response res) {
455       QueryParamsMap qm = req.queryMap();
456       String coord = prepInput(qm.value("NNCord"));
457       String printResult = "";
458 
459       StringTokenizer st = new StringTokenizer(coord, ",");
460       int inputSize = st.countTokens();
461 
462       int k = Integer.parseInt(qm.value("Number"));
463       if (k == 0) { // Not looking for neighbors.
464         printResult = ("0 Neighbors specified.");
465         Map<String, Object> variables = ImmutableMap.of("title",
466             "Stars: Query the database", "db", db, "result", printResult);
467         return new ModelAndView(variables, "query.ftl");
468       }
469 
470       if (inputSize == ARG_CORD_WEB) { // Coordinates specified
471         double x = Double.parseDouble(st.nextToken());
472         double y = Double.parseDouble(st.nextToken());
473         double z = Double.parseDouble(st.nextToken());
474         printResult = neighbors(k, x, y, z);
475       } else if (inputSize == ARG_NAME_WEB) { // Name specified
476         String name = st.nextToken();
477         String revertName = name.replaceAll("/////", " ");
478         printResult = neighborsName(k, revertName);
479       } else {
480         printResult = "Incorrect # of args for neighbors!";
481       }
482 
483       Map<String, Object> variables = ImmutableMap.of("title",
484           "Stars: Query the database", "db", db, "result", printResult);
485       return new ModelAndView(variables, "query.ftl");
486     }
487   }

529   private static class ExceptionPrinter implements ExceptionHandler {
530     @Override
531     public void handle(Exception e, Request req, Response res) {
532       res.status(EXCEPTION_PRINTER_THING);
533       StringWriter stacktrace = new StringWriter();
534       try (PrintWriter pw = new PrintWriter(stacktrace)) {
535         pw.println("<pre>");
536         e.printStackTrace(pw);
537         pw.println("</pre>");
538       }
539       res.body(stacktrace.toString());
540     }
