3   import java.util.Comparator;
4   
5   /**
6    * Comparator that sorts stars by a dimension x,y,or z represented by curDimen.
7    * @author cdluo
8    */
9   public class StarComparator implements Comparator<Star> {
10  
11    private int curDimen;
12  
13    /**
14     * Needed a constructor to set curDimen.
15     * @param cDimen
16     *          the Dimen to compare to.
17     */
18    public StarComparator(int cDimen) {
19      curDimen = cDimen;
20    }
21  
22    /**
23     * Compares stars based on the based in dimension.
24     * @param s1 1st star to be compared
25     * @param s2 2nd star to be compared
26     * @return int for sorting
27     */
28    @Override
29    public int compare(Star s1, Star s2) {
30      double s1Dimen = s1.getDimen().getDimenX(curDimen);
31      double s2Dimen = s2.getDimen().getDimenX(curDimen);
32  
33      if (s1Dimen > s2Dimen) {
34        return 1;
35      } else if (s1Dimen == s2Dimen) {
36        return 0;
37      } else {
38        return -1;
39      }
40    }
41  }
